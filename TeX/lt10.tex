\begin{definition} Рассмотрим точку $p$ на многообразии $M$ с аффинной связностью $\nabla$. Для любого вектора  $v \in T_p M$ существует единственная геодезическая $\gamma_v(t)$ такая, что \[ \gamma_v(0) = p, \qquad \dot{\gamma}_v(0) = v.\]

\textbf{Экспоненциальное отображение}  --- это отображение, которое сопоставляет касательному вектору $v \in T_P M$  точку $\gamma_v(1)$, т.е. \[ \exp_P (v)={\displaystyle \gamma _{v}(1)}.\]

\end{definition}

\begin{problem} \label{Prob:NormCoordZeroCristoff} Пусть связность $\nabla$ симметрична. Доказать, что в нормальных координатах с центром в точке $p$ все символы Кристоффеля обращаются в $0$ в этой точке: \[\Gamma_{ij}^k (p) =0. \] \end{problem}




\begin{proof}[Решение Задачи~\ref{Prob:NormCoordZeroCristoff}] Для любого вектора $v$ прямая \[\gamma(t) = t \mathbf{v}\] является геодезической. Уравнение геодезических для этих прямых имеет вид \[ \Gamma^k_{ij} (x) v^i v^j = 0. \]  Все эти уравнения одновременно выполняются в начале координат для всех векторов $v$. Таким образом при каждом фиксированном $k$ числа $\Gamma^k_{ij}(0)$ задают нулевую квадратичную форму\footnote{Здесь мы используем симметричность $\Gamma^k_{ij}$ по нижним индексам.}, и, следовательно, все они равны нулю. Задача~\ref{Prob:NormCoordZeroCristoff} решена.\end{proof}


%Доказать, что в нормальных координатах все прямые, проходящие через начало координат, ортогональны всем сферам с центром в начале координат. 

\begin{problem}[Лемма Гаусса] \label{Prob:GaussLemma}  Доказать, что для любой точки $p$ риманова многообразия $M$ образ любой достаточно малой сферы в $T_p M$ под действием экспоненциального отображения $\mathrm{exp}_p$ ортогонален всем геодезическим, проходящим через точку $p$. 
\end{problem}


\textit{Короткая формулировка леммы Гаусса:} Геодезическая сфера ортогональна ко всем геодезическим 
радиусам.


\begin{proof}[Решение Задачи~\ref{Prob:GaussLemma}] Рассмотрим сферу радиуса $r$ в касательном пространстве \[ S_{r}  \subset T_p M\] с центром в начале координат. Рассмотрим отображение \[ F(s, t) = \exp_p (tv(s)), \] где $v(s) \in S_{r}$. Тогда вектора \[F_t = \frac{\partial F}{\partial t}, \qquad F_s = \frac{\partial F}{\partial s}\] будут касательными к геодезическому радиусу и геодезической сфере соответственно. Нужно доказать, что \[ \left( F_t, F_s\right) =0. \] Это равенство очевидно выполнено при $t=0$, поскольку $F(s, 0) = p$. Поэтому остаётся доказать, что \[ \frac{d}{dt}\left( F_t, F_s\right)  = 0. \] Будем доказывать это равенство в точке $(s_0, t_0)$. Обозначим 

\begin{itemize}

\item $\gamma(t) = F(s_0, t)$ --- кривая при фиксированном $s =s_0$,

\item $\delta(t) = F(s, t_0)$  --- кривая при фиксированном $t =t_0$,

\item $\nabla_t = \nabla_{\dot{\gamma}(t)}$ и $\nabla_s = \nabla_{\dot{\delta}(s)}$.

\end{itemize} 

В таком случае \[ \frac{d}{dt}\left( F_t, F_s\right)  = \nabla_t \left( F_t, F_s\right). \] Так как связность согласована с метрикой \[ \nabla_t \left( F_t, F_s\right)  =\left( \nabla_t  F_t, F_s\right) +\left( F_t, \nabla_t  F_s\right).   \] Так как $\gamma(t)$ --- геодезическая,  \[ \nabla_t  F_t=0. \] 


\begin{assertion} \label{A:GaussLemmaSym} Для симметричной связности \[ \nabla_s  F_t=\nabla_t  F_s. \] \end{assertion} 

\begin{proof}[Доказательство Утверждения~\ref{A:GaussLemmaSym}] Достаточно явно написать выражения в локальных координатах:  \begin{gather*} \left( \nabla_s  F_t \right)^i = \frac{\partial^2 F^i}{\partial s \partial t} + \Gamma^i_{jk} \frac{\partial F^j}{\partial s} \frac{\partial F^k}{\partial t}, \\ \left( \nabla_s  F_t \right)^i = \frac{\partial^2 F^i}{\partial s \partial t} + \Gamma^i_{jk} \frac{\partial F^j}{\partial t} \frac{\partial F^k}{\partial s}. \end{gather*} Очевидно, что выражения равны, если $\Gamma^i_{jk}= \Gamma^i_{kj}$. Утверждение~\ref{A:GaussLemmaSym} доказано. \end{proof}

Переставляя индексы местами и вынося ковариантную производную за скобки, получаем \[  \nabla_t \left( F_t, F_s\right) =\left( F_t, \nabla_s F_t\right) = \frac{1}{2} \nabla_s  \left( F_t, F_t\right) = \frac{1}{2} \frac{d}{ds} \left\| v\right\|^2 =0,\] где последнее равенство выполнено, так как все вектора $v$ лежат на сфере фиксированного радиуса. Задача~\ref{Prob:GaussLemma}  решена. \end{proof}

\section{Полугеодезические координаты}


Координаты $(u^1,\dots, u^n)$  в окрестности точки риманова многообразия $P \in M^n$ называются \textbf{полугеодезическими}, если в них метрика имеет вид \begin{equation} \label{Eq:HalfGeodMetr} ds^2 = (du^1)^2 + \sum_{i, j \geq 2} g_{ij} du^i du^j.\end{equation}


\begin{problem} \label{Prob:HalfGeodMetric} Показать, что в полугеодезических координатах   $(u^1,\dots, u^n)$  первые координатные линии \[u^2= \mathrm{const}, \dots, u^n =\mathrm{const}\] являются геодезическими, ортогональными поверхностям уровня $u^1 = \mathrm{const}$.\end{problem}


\begin{proof}[Решение Задачи~\ref{Prob:HalfGeodMetric}] Утверждение про ортогональность очевидно. Чтобы доказать, что координатные линии \[\gamma(t) = \left(t, u^2_0, \dots, u^n_0\right) \] являются геодезическими, достаточно доказать, что $\Gamma^i_{11}=0,  i=1, \dots, n$. В данном случае \[ \Gamma^i_{11} = \frac{g^{i\alpha}}{2} \left( \frac{\partial g_{1 \alpha}}{\partial u^1} + \frac{\partial g_{\alpha 1}}{\partial u^1} - \frac{\partial g_{1 1}}{\partial u^{\alpha}} \right) = - \frac{g^{i\alpha}}{2} \frac{\partial 1}{\partial u^{\alpha}} =0. \] Задача~\ref{Prob:HalfGeodMetric} решена. \end{proof}

\begin{theorem} В окрестности любой точки риманова многообразия существуют полугеодезические координаты.    \end{theorem}

 \end{problem}
 
 
 \begin{proof}[Решение Задачи~\ref{Prob:GeodLocShort}] Рассмотрим полугеодезические координаты из Задачи \ref{Prob:HalfGeodFirstCoord}, в которых $\gamma(t)$ --- первая координатная линия. Без ограничения общности $u^1(x_2)>u^1(x_1)$. Тогда длина любой другой кривой $\hat{\gamma} = (u^1(t), \dots, u^n(t)$) удовлетворяет оценкам     \begin{gather*} l = \int_{\hat{\gamma} } \sqrt{ (\dot{u}^1)^2 + \sum_{i, j \geq 2} g_{ij} \dot{u}^i \dot{u}^j} dt \geq \\ \geq \int_{\hat{\gamma} } \left\| \dot{u}^1 \right\| dt \geq \int_{\hat{\gamma} } \dot{u}^1 dt = u^1(x_2)-u^1(x_1), \end{gather*} причем равенство возможно только при $u^2= \dots = u^n = 0$ и $u^1>0$, т.е. для отрезка геодезической $\gamma(t)$.  Задача~\ref{Prob:GeodLocShort} решена. \end{proof}

\begin{proof}[Решение Задачи~\ref{Prob:ShortGeod}] Кратчайшая является локально кратчайшей. Локально кратчайшие --- геодезические. Локально геодезическая --- геодезическая. Задача~\ref{Prob:ShortGeod} решена. 
\end{proof}

\section{Изометрии}


\begin{problem}\label{Prob:Isometry} Доказать,что 

\begin{itemize}

\item любая изометрия плоскости является аффинным преобразованием \[ x \to Ax + b, \qquad A \in \mathrm{O} (n), \]

 

\item и что любая изометрия сферы $S^n \subset \mathbb{R}^{n+1}$ является ограничением некоторого ортогонального преобразования $\mathbb{R}^{n+1}$.

\end{itemize}
   \end{problem}
   
   
   
   Для решения Задачи \ref{Prob:Isometry} можно воспользоваться следующим утверждением.


\begin{theorem} Если изометрия связного риманова многообразия $f: M^n \to M^n$ оставляет на месте точку $ f (P) = P$ и тождественно действует на соответствующем касательном пространстве \[ df |_P = \mathrm{id}: T_P M \to T_P M, \] то она является тождественным преобразованием $f = \mathrm{id}$. \end{theorem}

Это немедленное следствие того факта, что через любую точку в любом направлении проходит единственная геодезическая. 




\begin{corollary} Любая изометрия (связных) римановых многообразий \[f: M_1 \to M_2\] полностью определяется изометричным отображением одного касательного пространства \[df : T_P M_1 \to T_{f(P)} M_2.\] \end{corollary}

\begin{proof}[Решение Задачи~\ref{Prob:Isometry}] Легко видеть, что указанные преобразования реализуют всевозможные изометрии произвольного касательного пространства.  Задача~\ref{Prob:Isometry} решена. \end{proof}
